## **EXAMPLE INTEGRATION IN BOLT PROMPT**

Add this to your Bolt prompt:

```javascript
// Harsh Quotes Array
const harshQuotes = [
  "Your competition is training while you make excuses.",
  "Every time you fold, you train yourself to be weak.",
  "The gap between who you claim to be and who you are IS your life.",
  // ... all 30 harsh quotes
];

// Display random quote on dashboard
function RotatingQuote() {
  const [quote, setQuote] = useState(harshQuotes[0]);
  
  useEffect(() => {
    const interval = setInterval(() => {
      const randomQuote = harshQuotes[Math.floor(Math.random() * harshQuotes.length)];
      setQuote(randomQuote);
    }, 30000); // Change every 30 seconds
    
    return () => clearInterval(interval);
  }, []);
  
  return (
    <div className="harsh-quote">
      <p>{quote}</p>
    </div>
  );
}
```
