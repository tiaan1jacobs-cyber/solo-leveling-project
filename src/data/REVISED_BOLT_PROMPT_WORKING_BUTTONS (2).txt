// Award XP function
function awardXP(stat, amount) {
  setTodayXP(todayXP + amount);
  
  // Update stat
  setUserData(prev => {
    const newXP = prev.stats[stat].currentXP + amount;
    const xpNeeded = prev.stats[stat].level * 100;
    
    // Check for level up
    if (newXP >= xpNeeded) {
      const remainingXP = newXP - xpNeeded;
      const newLevel = prev.stats[stat].level + 1;
      
      // Show level up animation
      showLevelUp(stat, newLevel);
      
      return {
        ...prev,
        stats: {
          ...prev.stats,
          [stat]: {
            level: newLevel,
            currentXP: remainingXP,
            totalXP: prev.stats[stat].totalXP + amount
          }
        }
      };
    }
    
    // No level up, just add XP
    return {
      ...prev,
      stats: {
        ...prev.stats,
        [stat]: {
          ...prev.stats[stat],
          currentXP: newXP,
          totalXP: prev.stats[stat].totalXP + amount
        }
      }
    };
  });
  
  // Save immediately
  localStorage.setItem('apexData', JSON.stringify(userData));
}
