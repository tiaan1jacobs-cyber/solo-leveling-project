function RotatingHarshQuote() {
  const [currentQuote, setCurrentQuote] = useState(harshQuotes[0]);
  
  useEffect(() => {
    const interval = setInterval(() => {
      const randomIndex = Math.floor(Math.random() * harshQuotes.length);
      setCurrentQuote(harshQuotes[randomIndex]);
    }, 30000); // Every 30 seconds
    
    return () => clearInterval(interval);
  }, []);
  
  return (
    <div className="harsh-quote-container">
      <p className="harsh-quote">{currentQuote}</p>
    </div>
  );
}

export default RotatingHarshQuote;
